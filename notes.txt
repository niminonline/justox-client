import { Router } from '@angular/router';
constructor(private router: Router) {}
refreshPage() {
  const currentUrl = this.router.url;
  this.router.navigateByUrl('/', { skipLocationChange: true }).then(() => {
    this.router.navigateByUrl(currentUrl); });
}
